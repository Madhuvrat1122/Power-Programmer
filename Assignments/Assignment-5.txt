def dreams(N,A):
    res = 0
    A = sorted(A)
    while(N>3):
        op1 = A[0]+2*A[1]+A[N-1]
        op2 = 2*A[0]+A[N-1]+A[N-2]
        res += min(op1,op2)
        N -= 2
    if N==3:
        res += sum(A[:3])
    elif N==2:
        res += A[1]
    else:
        res += A[0]
    return res

#Main Code
T = int(input())
for i in range(T):
    N = int(input())
    A = list(map(int,input().split(" ")))
    print(dreams(N,A))
